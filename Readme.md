

<h1> Install</h1> 
	
	npm i safe_maths

<h1> Import</h1> 
	
	const safeMaths = require('safe_maths');

<h1> Use</h1>
	
	safeMaths.add(1,2); //result 3<br>
	safeMaths.mul(1,2); //result 2<br>
	safeMaths.pow(1,2); //result 1<br>
	safeMaths.div(1,2); //result 0.5<br>
	safeMaths.sub(1,2); //result -1<br>
